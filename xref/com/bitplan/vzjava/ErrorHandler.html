1   package com.bitplan.vzjava;

3   import java.io.PrintWriter;
4   import java.io.StringWriter;
5   import java.util.logging.Level;
6   import java.util.logging.Logger;
7   
8   /**
9    * simple ErrorHandler
10   * @author wf
11   *
12   */
13  public class ErrorHandler {
14    protected static Logger LOGGER = Logger.getLogger("com.bitplan.nonolet");
15    
16    /***
17     * handle the given Throwable
18     * 
19     * @param th
20     */
21    public static void handle(Throwable th) {
22      LOGGER.log(Level.WARNING, "Error " + th.getClass().getName()+ ":"+ th.getMessage());
23      StringWriter sw = new StringWriter();
24      th.printStackTrace(new PrintWriter(sw));
25      LOGGER.log(Level.WARNING, "Stacktrace: " + sw.toString());
26    }
27  
28    /**
29     * issue a warning
30     * @param msg
31     */
32    public static void warn(String msg) {
33      LOGGER.log(Level.WARNING,msg);
34    }
35  }
