21  package com.bitplan.vzjava.rest;
22  import com.bitplan.rest.RestServerImpl;
23  import com.bitplan.rest.providers.JsonProvider;
24  import com.bitplan.rest.users.UserManagerImpl;
25  
26  /**
27   * RESTful Server for Volksz√§hler
28   */
29  public class VZServer extends RestServerImpl{
30  
31    /**
32     * construct VZServer
33     * setting defaults
34     * @throws Exception 
35     */
36    public VZServer() throws Exception {
37      settings.setHost("0.0.0.0");
38      settings.setPort(8380);
39      
40      settings.setContextPath("/vz".toLowerCase());
41      settings.addClassPathHandler("/", "/static/");
42      String packages="com.bitplan.vzjava.resources;com.bitplan.vzjava.rest;com.bitplan.rest.providers";
43      JsonProvider.registerType(UserManagerImpl.class);
44      settings.setPackages(packages);
45    }       
46         
47    /**
48     * start Server
49     * 
50     * @param args
51     * @throws Exception
52     */
53     public static void main(String[] args) throws Exception {
54       VZServer rs=new VZServer();
55       rs.settings.parseArguments(args);
56       rs.startWebServer();
57     } // main
58  } // NonoletServer
